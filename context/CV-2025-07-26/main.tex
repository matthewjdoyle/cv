\newenvironment{twocolentry}[2][]{
    \onecolentry
    \def\secondColumn{#2}
    \setcolumnwidth{\fill, 3.0 cm}
    \begin{paracol}{2}
}{
    \switchcolumn \raggedleft \secondColumn
    \end{paracol}
    \endonecolentry
} % new environment for two column entries

\newenvironment{threecolentry}[3][]{
    \onecolentry
    \def\thirdColumn{#3}
    \setcolumnwidth{1.5 cm, \fill, 4.5 cm}
    \begin{paracol}{3}
    {\raggedright #2} \switchcolumn
}{
    \switchcolumn \raggedleft \thirdColumn
    \end{paracol}
    \endonecolentry
} % new environment for three column entries

\newenvironment{techcolentry}[2][]{
    \onecolentry
    \def\secondColumn{#2}
    \setcolumnwidth{4.2 cm, \fill}
    \begin{paracol}{2}
}{
    \raggedleft \switchcolumn \raggedright \secondColumn
    \end{paracol}
    \endonecolentry
} % new environment for technology entries with wider right column
